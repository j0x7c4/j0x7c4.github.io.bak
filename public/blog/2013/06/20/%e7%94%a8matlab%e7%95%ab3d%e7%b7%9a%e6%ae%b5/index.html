用matlab畫3D線段

figure %create the figure
t = 0:1:135  %create the vector
h1 = plot3(t,t,135-t)  %draw x-y-z+135 = 0
hold on
grid on
h2 = plot3(t,t,zeros(length(t))) %draw x-y = 0
set(h1,'color',[0,0,0])  %set color black
set(h2,'color',[0,0,0])
